    def save(self):
        models = self.all()
        first_index = 1
        if self.__dict__.get('id') is None:
            if len(models) > 0:
                self.id = models[-1].id + 1
            else:
                self.id = first_index
            models.append(self)
        else:
            index = -1
            for i, m in enumerate(models):
                if m.id == self.id:
                    index = i
                    break
            if index > -1:
                models[index] = self
        l = [m.__dict__ for m in models]
        path = self.db_path()
        save(l, path)
